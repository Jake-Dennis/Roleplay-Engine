```typescript
function parseJsonOrSplit(val: string | null): string[] {
  if (!val) return [];
  if (val.startsWith("[")) {
    try {
      const parsed = JSON.parse(val);
      if (Array.isArray(parsed)) return parsed.map(String);
    } catch {
      // Fall through to comma-split
    }
  }
  return val.split(",").map((s) => s.trim()).filter(Boolean);
}
```
- Handles JSON arrays (val.startsWith("[")): YES
- Validates Array.isArray after parse: YES
- Maps to String for type safety: YES
- Falls back to comma-split on parse error: YES
